		<script>
		
		
		window.setTimeout(function(){
			
			// fonte
			var cc = $("#Ficheiro3").hide();
			$("#Fonte_amostras input").bind('change', function(){
				var check = false;
				$("#Fonte_amostras input").each(function(){
					var val = $(this).val(); 
					if (val == "CrioBio HCB" || val == "CrioBio HCB/Unesp") {
						if ($(this).is(":checked")) {
						 check = true;							 
						}
					}
				   check ? cc.show() : cc.hide();
				});
			});
			
			
			var om = $("#Outro_metodo_processamento_material").hide();
			$("#Metodo_processamento select").bind('change', function(){
				var value = $(this).val();
				if (value == "Outro") {
					om.show();
				} else {
					om.hide().find('textarea').val('');					
				}
			});
			
		}, 1000);
