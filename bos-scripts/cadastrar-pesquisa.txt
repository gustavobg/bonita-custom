		function adicionarCampos(idBotaoAdicionar, classRow, td) {
			//var classRow = $(classRow);
			var botaoAdicionar = $(idBotaoAdicionar).find("button");
			//var inputs = camposContainer.find("input");
			
			
			var inputs = $("." + classRow + " td:nth-child(" + td + ")").find("input"); 
			inputs.each(function(){
				$(this).parents(".bonita_form_entry").hide();
			});
			
			var camposCount = inputs.length; 
			var count = 0;
			
			
			botaoAdicionar.bind("click", function(){
				if (count <= camposCount) {
					$(inputs[count]).parents(".bonita_form_entry").show();
					count++;
				}
			});
			
		}
		
		function adicionarCamposSelect(idBotaoAdicionar, classRow, td) {
			//var classRow = $(classRow);
			var botaoAdicionar = $(idBotaoAdicionar).find("button");
			//var inputs = camposContainer.find("input");
			
			
			var inputs = $("." + classRow + " td:nth-child(" + td + ")").find("select"); 
			inputs.each(function(){
				$(this).parents(".bonita_form_entry").hide();
			});
			
			var camposCount = inputs.length; 
			var count = 0;
			
			
			
			botaoAdicionar.bind("click", function(){
				if (count <= camposCount) {
					$(inputs[count]).parents(".bonita_form_entry").show();					
					count++;
				}
			});
			
		}
		
		function adicionarOutras(idWidget, idOutroWidget) {
			
			var outro = $(idOutroWidget); 				
			var txtarea = $('<textarea class="bonita_form_field bonita_form_field_right outros"></textarea>');
			
			txtarea.appendTo($(idWidget).parents(".group-widgets")).hide();
			
			txtarea.bind("keyup", function(){								
				outro.find("input").val($(this).val());					
			});
			
			$(idWidget).bind("change", function(){
				val = $(this).val();
				val = val.toLowerCase();
				
				if ($(this).is(":checked")) {
					if (val == "outros" || val == "outras" || val == "outro" || val == "outra") {						
						txtarea.show();
					} 
				} else {					
					outro.find("input").val("");						
					txtarea.hide().val("");			
					
				}
			})
		}
		
		function adicionarOutro(idWidget, idOutroWidget) {
					
					var outro = $(idOutroWidget);
					outro.hide();
					
					$(idWidget).bind("change", function(){
						val = $(this).val();
						val = val.toLowerCase();
						
						if ($(this).is(":checked")) {
							if (val == "outros" || val == "outras" || val == "outro" || val == "outra") {						
								outro.show();
							} 
						} else {					
							outro.hide().find("input").val("");						
									
							
						}
					})
		}

					var maxChar = elemento.data("maxChar");		

			        elemento.keyup(function() {
			            var text = elemento.val();
			            var textlength = text.length;            
			            var wrap = elemento.parent('.ui-limite-wrap');   
							

			            if (textlength > maxChar) {
			                $(wrap).children('.ui-limite-info').html('').html('Você não pode digitar mais que ' + maxChar + ' caracteres');
			                elemento.val(text.substr(0, maxChar));
			                return false;
			            }
			            else {                
			                var caracterDisponivel = maxChar - textlength;
			                var caracter = ((caracterDisponivel > 1) ? 'caracteres' : 'caractere');
			                $(wrap).children('.ui-limite-info').html('').html('Você pode digitar mais ' + caracterDisponivel + ' ' + caracter);
			                return true;
			            }
			        })					
				});
				
				
				
				
				
			}, 1000);
