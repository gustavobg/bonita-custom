		<script type="text/javascript">
		var arrEls = [];
		var arrPop = [];
		
		function adicionarCampos(idBotaoAdicionar, classRow, td) {
			//var classRow = $(classRow);
			var botaoAdicionar = $(idBotaoAdicionar).find("button");
			//var inputs = camposContainer.find("input");
			
			
			var inputs = $("." + classRow + " td:nth-child(" + td + ")").find("input"); 
			inputs.each(function(){
				$(this).parents(".bonita_form_entry").hide();
			});
			
			var camposCount = inputs.length; 
			
			
			
			botaoAdicionar.bind("click", function(){
				if (camposCount > 0) {
					$(inputs[camposCount - 1]).parents(".bonita_form_entry").show();
					camposCount--;
				}
			});
			
		}
		
		function adicionarOutras(idWidget, idOutroWidget) {
			
			var outro = $(idOutroWidget); 				
			var txtarea = $('<textarea class="bonita_form_field bonita_form_field_right outros"></textarea>');
			
			txtarea.appendTo($(idWidget).parents(".group-widgets")).hide();
			
			txtarea.bind("keyup", function(){								
				outro.find("input").val($(this).val());					
			});
			
			$(idWidget).bind("change", function(){
				val = $(this).val();
				val = val.toLowerCase();
				
				if ($(this).is(":checked")) {
					if (val == "outros" || val == "outras" || val == "outro" || val == "outra") {						
						txtarea.show();
					} 
				} else {					
					outro.find("input").val("");						
					txtarea.hide().val("");			
					
				}
			})
		}
		
		$(document).ready(function(){
			// pega elementos				
			$("#Caixa_de_sugestao1, #Caixa_de_sugestao2, #Caixa_de_sugestao3, #Caixa_de_sugestao4").each(function(){
				// esconde todos
				var td = $(this).parent("td").css('visibility','hidden');
				arrEls.unshift([$(this), td]);
			});
			
			$("#AdicionarPesquisador").bind('click', function(){
				if (arrEls.length > 0) {
					var el = arrEls.pop();
					el[1].css('visibility','visible');
					arrPop.push(el);
				}
			});
			
			$("#RemoverPesquisador").bind('click', function(){
				if (arrPop.length > 0) {
					var el = arrPop.pop();
					el[1].css('visibility','visible');
				}
			});			
			
			// Campos de texto
			window.setTimeout(function(){
				
				// Novo centro pesquisa				
				
				// Exibe CPOM
				var linhaCPOM = $("#Linha_CPOM").hide();
				$("#Linha_pos_graduacao input").bind('change', function(){
					var exibeCPOM = false;	
					$("#Linha_pos_graduacao input").each(function(){
						var val = $(this).val();
						if (val == "1-Biologia Tumoral" || val == "3-Epidemiologia Clínica e Molecular em Oncologia" || val == "4-Fatores Ambientais e Câncer")	{
							if ($(this).is(":checked")) {
								exibeCPOM = true;
							}
						}
						exibeCPOM ? linhaCPOM.show() : linhaCPOM.hide();
					});					
				});
				
				// Multicentrico
				var cc = $("#Centro_coordenador").hide();
				$("#Centros_pesquisa_envolvidos input").bind('change', function(){
					var check = false;
					$("#Centros_pesquisa_envolvidos input").each(function(){
						var val = $(this).val(); 
						if (val == "Multicêntrico" || val == "Coparticipante") {
							if ($(this).is(":checked")) {
							 check = true;							 
							}
						}
					   check ? cc.show() : cc.hide();
					});
				});
				
				// Outros - Área de interesse
				adicionarOutras("#gwt-uid-69", "#Outros_area_interesse");
				adicionarOutras("#gwt-uid-58", "#Outros_areas_de_saude");
				
				// Campos
				adicionarCampos("#AdicionarPesquisador", "row_pesq", 1);
				adicionarCampos("#Botao_simples1", "row_pesq", 2);
				
				adicionarCampos("#AdicionarDepartamentoEnvolvido", "row_dep", 1);
				adicionarCampos("#Botao_simples2", "row_dep", 2);
				
				
				$("textarea[data-max-char]").each(function(){	
					
					var elemento = $(this);

			        elemento.wrap('<div class="ui-limite-wrap"></div>').parent('.ui-limite-wrap').append('<div class="ui-limite-info"></div>');
					
					var maxChar = elemento.data("maxChar");		

			        elemento.keyup(function() {
			            var text = elemento.val();
			            var textlength = text.length;            
			            var wrap = elemento.parent('.ui-limite-wrap');   
							

			            if (textlength > maxChar) {
			                $(wrap).children('.ui-limite-info').html('').html('Você não pode digitar mais que ' + maxChar + ' caracteres');
			                elemento.val(text.substr(0, maxChar));
			                return false;
			            }
			            else {                
			                var caracterDisponivel = maxChar - textlength;
			                var caracter = ((caracterDisponivel > 1) ? 'caracteres' : 'caractere');
			                $(wrap).children('.ui-limite-info').html('').html('Você pode digitar mais ' + caracterDisponivel + ' ' + caracter);
			                return true;
			            }
			        })					
				});
				
				
				
				
				
			}, 1000);			
		});
