			<script type="text/javascript">
			function exibeOcultaSelect(idSelect, divs) {
				var divs = $(divs);
				divs.hide();
				$(idSelect).find("select").bind("change",function(){
					var val = $(this).val();
					val = val.toLowerCase();					
					if (val == "sim" || val == 1) {
						divs.show();
					} else {
						divs.hide();
					}
				});
			}
			
			$(document).ready(function(){
				window.setTimeout(function(){
					exibeOcultaSelect("#Necessita_amostra_biologica", "#Tipos_de_amostras, #Quantidade_amostras");
					exibeOcultaSelect("#Encaminha_amostra", "#Instituicao_encaminha_amostra");
					exibeOcultaSelect("#Conflito_interesse", "#Qual_conflito_interesse");	
				},1000);
			});
